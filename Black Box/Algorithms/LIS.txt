void doLIS(const vector<ll> &S, vector<ll> &LIS) {
    vector<ll> L(S.size());
    ll lisCount = 0;
    for (size_t i = 0; i < S.size(); ++i)
    {
        // if you need equal values change it to upper_bound
        ll pos = lower_bound(L.begin(), L.begin() + lisCount, S[i]) - L.begin();
        L[pos] = S[i];
        if (pos == lisCount)
            ++lisCount;
        LIS[i] = pos + 1;
    }
}
