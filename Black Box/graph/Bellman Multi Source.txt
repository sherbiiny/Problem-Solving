struct Edge {
    int u, v, c;
    Edge(int U, int V, int C) {
        u = U,
        v = V,
        c = C;
    }
};

int n, m;
bool cycle = false;
vector<Edge> edges;

vector<ll> bellmanMultisource(int root) {
    vector<ll> dist(n + 1, 0);

    bool flag = true;
    int cnt = 0; // if cnt > n - 1 there is a negative cycle
    while(flag) {
        cnt++;
        flag = false;
        for(Edge e : edges)
            if(dist[e.u] > dist[e.v] + e.c)
                dist[e.u] = dist[e.v] + e.c, flag = true;

        if(cnt > n) {
            cycle = true;
            break;
        };
    }
    return dist;
}
